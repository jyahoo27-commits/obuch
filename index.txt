    """
    Вычисляет точность модели.
    
    Args:
        y_true: список реальных меток
        y_pred: список предсказанных меток

        float: точность от 0 до 1

<h2>ООП (Объектно-ориентированное программирование)</h2>
<p>В DS-проектах классы используются для создания моделей, пайплайнов и кастомных трансформеров данных. Scikit-learn целиком построен на ООП:</p>

<pre class="code-block"><code">class DataPreprocessor:
    """
    Препроцессор данных для банковских скоринговых задач.
    Пример класса в DS-стиле.
    """
    
    def __init__(self, fillna_strategy='median'):
        self.fillna_strategy = fillna_strategy
        self.stats = {}  # сохраняем статистику при обучении
        self.is_fitted = False
    
    def fit(self, data):
        """Обучаемся на тренировочных данных"""
        for col in data:
            if self.fillna_strategy == 'median':
                self.stats[col] = data[col].median()
            else:
                self.stats[col] = data[col].mean()
        self.is_fitted = True
        return self  # позволяет делать цепочки: .fit().transform()
    
    def transform(self, data):
        """Применяем трансформацию"""
        if not self.is_fitted:
            raise ValueError("Сначала вызови fit()!")
        result = data.copy()
        for col, fill_value in self.stats.items():
            if col in result.columns:
                result[col] = result[col].fillna(fill_value)
        return result
    
    def fit_transform(self, data):
        """Удобный метод: fit + transform"""
        return self.fit(data).transform(data)

# Использование
preprocessor = DataPreprocessor(fillna_strategy='median')
clean_data = preprocessor.fit_transform(raw_data)</code></pre>

    """Читает транзакции по одной — экономия памяти"""
    with open(filepath, 'r') as f:
        for line in f:
            yield json.loads(line)  # yield вместо return

# Пример из Сбера:

<h2>Практический размер эффекта (Cohen's d)</h2>

<pre class="code-block"><code># Статистическая значимость ≠ практическая значимость!
# Пример: при 1 миллионе клиентов любая крошечная разница 
# будет "статистически значимой"

def cohens_d(group1, group2):
    """
    Размер эффекта по Коэну:
    d < 0.2 — очень малый эффект
    0.2-0.5 — малый эффект  
    0.5-0.8 — средний эффект
    d > 0.8 — большой эффект
    """
    n1, n2 = len(group1), len(group2)
    pooled_std = np.sqrt(((n1-1)*np.var(group1,ddof=1) + 
                          (n2-1)*np.var(group2,ddof=1)) / (n1+n2-2))
    return (np.mean(group1) - np.mean(group2)) / pooled_std

d = cohens_d(group_b, group_a)
print(f"Cohen's d = {d:.3f}")
# Интерпретация: малый (~0.2), средний (~0.5), большой (>0.8) эффект</code></pre>

<h2>Underfitting vs Overfitting — главный баланс в ML</h2>

<pre class="code-block"><code">"""
UNDERFITTING (недообучение):
- Модель слишком простая
- Плохо работает и на train, и на test
- Решение: усложни модель, добавь признаки

OVERFITTING (переобучение):
- Модель слишком сложная, "запомнила" train данные
- Отлично на train, плохо на test
- Решение: регуляризация, больше данных, упрощение модели

ПРИМЕР:
- Train accuracy: 99%, Test accuracy: 62% → OVERFITTING
- Train accuracy: 65%, Test accuracy: 63% → UNDERFITTING  
- Train accuracy: 87%, Test accuracy: 85% → ХОРОШИЙ БАЛАНС

Bias-Variance Tradeoff:
- Высокий bias → underfitting
- Высокая variance → overfitting
- Нужен баланс (sweet spot)
"""</code></pre>

<div class="key-point">
  <span class="key-icon">🎯</span>
  <p><strong>Связь с твоим опытом разметки:</strong> Когда ты проверял качество разметки для обучения нейросетей — ты напрямую влиял на качество y_train. Плохая разметка → плохой y_train → неверно обученная модель. Теперь ты понимаешь почему контроль качества разметки так критичен!</p>

<h2>Precision, Recall, F1</h2>

<pre class="code-block"><code>"""
Precision (Точность) = TP / (TP + FP)
"Из всех, кого мы назвали мошенниками — сколько реально ими были?"
Высокий Precision → мало ложных тревог

Recall (Полнота) = TP / (TP + FN)  
"Из всех реальных мошенников — скольких мы поймали?"
Высокий Recall → мало пропущенных мошенников

F1-Score = 2 * (Precision * Recall) / (Precision + Recall)
Баланс между Precision и Recall

TRADE-OFF: Улучшая Precision, ухудшаешь Recall и наоборот!

ВЫБОР МЕТРИКИ ПО ЗАДАЧЕ:
- Антифрод → Recall важнее (дороже пропустить мошенника)
- Спам-фильтр → Precision важнее (дороже потерять важное письмо)
- Медицинский диагноз → Recall (дороже не выявить болезнь)
- Кредитный скоринг → F1 или AUC-ROC (нужен баланс)
"""

from sklearn.metrics import precision_score, recall_score, f1_score

precision = precision_score(y_true, y_pred)
recall = recall_score(y_true, y_pred)
f1 = f1_score(y_true, y_pred)

print(f"Precision: {precision:.2f}")
print(f"Recall:    {recall:.2f}")
print(f"F1-Score:  {f1:.2f}")</code></pre>

1. Берём случайную выборку с возвращением (bootstrap)
2. Обучаем дерево на этой подвыборке
3. Каждое дерево видит только random subset признаков
4. Итог = большинство голосов / среднее

Преимущества Random Forest:
✅ Устойчив к выбросам
✅ Мало гиперпараметров
✅ Работает "из коробки" без нормализации
✅ Хорошо показывает важность признаков
✅ Нет проблемы overfitting как у одного дерева

Недостатки:
❌ Медленнее на инференсе чем одно дерево
❌ Меньше точность чем у Gradient Boosting
"""

rf = RandomForestClassifier(
    n_estimators=200,       # количество деревьев
    max_depth=10,           # глубина каждого дерева
    max_features='sqrt',    # sqrt(n_features) признаков на каждом узле
    min_samples_leaf=5,
    n_jobs=-1,              # использовать все ядра CPU
    random_state=42
)
rf.fit(X_train, y_train)

# Оценка
from sklearn.metrics import roc_auc_score
y_proba = rf.predict_proba(X_test)[:, 1]
print(f"Random Forest AUC-ROC: {roc_auc_score(y_test, y_proba):.4f}")</code></pre>

<h2>Gradient Boosting — лучший алгоритм для табличных данных</h2>

<pre class="code-block"><code">"""
Gradient Boosting — последовательный ансамбль:
Каждое следующее дерево учится на ОШИБКАХ предыдущих!

Алгоритм:
1. F0(x) = среднее y (простое начальное предсказание)
2. r1 = y - F0(x)  (остатки — ошибки)
3. Обучаем дерево h1 на остатках r1
4. F1(x) = F0(x) + lr * h1(x)  (lr = learning rate, шаг)
5. r2 = y - F1(x)  (новые остатки)
6. Повторяем N раз...

Результат: Fn(x) = F0 + lr*h1 + lr*h2 + ... + lr*hn

CatBoost от Яндекса — лучший для:
✅ Категориальных признаков (встроенная обработка!)
✅ Быстрое обучение на CPU
✅ Хороший базовый результат без тюнинга
✅ Широко используется в Сбере, Яндексе, ВТБ
"""

# CatBoost
from catboost import CatBoostClassifier

# Определяем категориальные признаки
cat_features = ['region', 'product_type', 'employment_type']

model = CatBoostClassifier(
    iterations=1000,        # количество деревьев
    learning_rate=0.05,     # шаг (меньше = лучше, но медленнее)
    depth=6,                # глубина каждого дерева
    l2_leaf_reg=3,          # L2-регуляризация
    cat_features=cat_features,  # автообработка категорий!
    eval_metric='AUC',
    early_stopping_rounds=50,   # остановка если нет улучшения
    random_seed=42,
    verbose=100             # логирование каждые 100 итераций
)

model.fit(
    X_train, y_train,
    eval_set=(X_val, y_val),
    use_best_model=True
)

print(f"Best iteration: {model.best_iteration_}")
y_pred = model.predict_proba(X_test)[:, 1]
print(f"CatBoost AUC-ROC: {roc_auc_score(y_test, y_pred):.4f}")</code></pre>

<h2>Preprocessing текста</h2>

<pre class="code-block"><code">import re
import nltk
from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords
from nltk.stem import SnowballStemmer

nltk.download('punkt')
nltk.download('stopwords')

def preprocess_text(text):
    """Препроцессинг русского текста"""
    # 1. Нижний регистр
    text = text.lower()
    
    # 2. Удаление специальных символов
    text = re.sub(r'[^а-яёa-z0-9\s]', '', text)
    
    # 3. Токенизация
    tokens = word_tokenize(text, language='russian')
    
    # 4. Удаление стоп-слов
    stop_words = set(stopwords.words('russian'))
    tokens = [t for t in tokens if t not in stop_words and len(t) > 2]
    
    # 5. Стемминг (обрезка до основы)
    stemmer = SnowballStemmer('russian')
    tokens = [stemmer.stem(t) for t in tokens]
    
    return ' '.join(tokens)

# Пример
review = "Ужасное обслуживание! Сотрудники не помогли с проблемой."
processed = preprocess_text(review)
print(processed)  # ужасн обслуживан сотрудник помог проблем</code></pre>

<h2>Мониторинг моделей в продакшне</h2>

<pre class="code-block"><code">"""
Почему модели деградируют?

Data Drift — изменение распределения входных данных:
  - COVID изменил транзакционное поведение клиентов
  - Инфляция изменила "нормальный" уровень дохода

Concept Drift — изменение связи X→y:
  - Антифрод: мошенники адаптируются
  - Кредитный скоринг: меняются экономические условия

ЧТО МОНИТОРИРОВАТЬ:
1. Качество модели (AUC-ROC на свежих данных)
2. Распределение входных признаков (PSI метрика)
3. Распределение предсказаний
4. Латентность (время ответа API)
5. Бизнес-метрики (конверсия, дефолт-рейт)
"""

# PSI (Population Stability Index) — мониторинг drift
def calculate_psi(expected, actual, buckets=10):
    """
    PSI < 0.1: модель стабильна
    0.1 < PSI < 0.25: незначительные изменения, следить
    PSI > 0.25: значительный дрифт, переобучать модель!
    """
    def psi_bucket(e, a):
        if a == 0:
            a = 0.0001
        if e == 0:
            e = 0.0001
        return (e - a) * np.log(e / a)
    
    expected_pct = np.histogram(expected, buckets)[0] / len(expected)
    actual_pct = np.histogram(actual, buckets)[0] / len(actual)
    
    return sum(psi_bucket(e, a) for e, a in zip(expected_pct, actual_pct))</code></pre>

const LEVELS = [
  { level: 1, title: 'Стажёр', minXP: 0,    icon: '🌱' },
  { level: 2, title: 'Аналитик данных', minXP: 200,  icon: '📊' },
  { level: 3, title: 'Junior DS', minXP: 500,  icon: '🔬' },
  { level: 4, title: 'Middle DS', minXP: 1000, icon: '⚡' },
  { level: 5, title: 'Senior DS', minXP: 1800, icon: '🏆' },
  { level: 6, title: 'Lead DS', minXP: 2800, icon: '🌟' },
];

function getLevelInfo(xp) {
  let currentLevel = LEVELS[0];
  let nextLevel = LEVELS[1];
  
  for (let i = LEVELS.length - 1; i >= 0; i--) {
    if (xp >= LEVELS[i].minXP) {
      currentLevel = LEVELS[i];
      nextLevel = LEVELS[i + 1] || null;
      break;
    }
  }
  
  const progressToNext = nextLevel 
    ? ((xp - currentLevel.minXP) / (nextLevel.minXP - currentLevel.minXP)) * 100
    : 100;
    
  return { currentLevel, nextLevel, progressToNext };
}
